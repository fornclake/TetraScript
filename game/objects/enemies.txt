entity Stalfaux {
	Health 0.5
	Speed 40
	Damage 0.5
	Size 6, 7
	SpriteSheet "objects/stalfaux", 6, 1
	States {
		Spawn:
			anim_play("inanimate")
			trigger body_within(96,player) state Animate
		Animate:
			anim_play("animate")
			trigger anim_finished() state Active
		Active:
			anim_play("walk")
			wander()
			trigger timer(2) state BeginThrow
			trigger body_outside(96,player) state UnAnimate
		UnAnimate:
			anim_play("unanimate")
			trigger anim_finished() state Spawn
		BeginThrow:
			anim_play("throw")
			trigger timer(0.5) state Throw
		Throw:
			use_weapon("Bone")
			state Active
	}
	Animations {
		inanimate:
			0 0
			loop
		animate:
			0 0.5
			1 0.5
		unanimate:
			1 0.5
			0 0.5
		walk:
			2 0.25
			3 0.25
			loop
		throw:
			4 0.25
			5 0.25
	}
}
